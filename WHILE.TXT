// aqui se acumumula la edadades de la familia

var numeroMiembros= parseInt(prompt("ingrese la cantidad de mienbros de la familia"));
var contador = 1;
var totalEdades = 0;


 while(contador <= numeroMiembros){
 

edad = parseInt(prompt("ingrese la edad del familiar"))
	totalEdades = totalEdades + edad;
	contador ++ ;

 }
//promedio es lo mismo que media
mediaEdades= totalEdades/numeroMiembros;
